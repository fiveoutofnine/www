## With action

To add an action to the toast, pass in a `Button` element to the `action` prop. Note that `action` takes in any React node, but it is intended to be a `Button` element that triggers some relevant action.

<DesignComponentsDisplay>
  <ToastButton
    title="Contract deployed"
    description="Contract deployed to 0x00000000â€¦."
    intent="success"
    action={
      <Button
        size="sm"
        intent="success"
        href="https://etherscan.io/tx/0x0100b7a91f27db8ea706a0f729bf0677e45a8647cf7dd25fc249dce3710bea81"
        rightIcon={<ExternalLink />}
        children="View"
        newTab
      />
    }
    children="Show toast"
  />
</DesignComponentsDisplay>

export const ToastButton = ({ title = 'Title', description = 'Swipe right to remove.', intent, action, ...rest }) => {
  const { toast } = useToast();

return <Button intent={intent} onClick={() => toast({ title, description, intent, action })} {...rest} />;
}

export default (props) => (
  <DesignLayout name="Toast" selected="/design/component/toast" {...props} />
);
