export const ComponentFeature = ({ href, children }) => {
  const [isMounted, setIsMounted] = useState(false);
  useEffect(() => setIsMounted(true), []);
  const isMobile = isMounted ? /iPhone|iPad|iPod|Android/i.test(navigator.userAgent) : false;
  return (
    <div
      className="group relative flex h-32 w-full items-center justify-center rounded-xl border border-gray-7 bg-gray-2 transition-colors hover:border-gray-8"
      tabIndex={-1}
    >
      {children}
      <IconButton
        variant="ghost"
        className={clsx(
          'absolute bottom-2 right-2',
          isMobile ? 'flex' : 'hidden animate-in fade-in group-hover:flex',
        )}
        href={href}
      >
        <ArrowRight />
      </IconButton>
    </div>
  );
};
